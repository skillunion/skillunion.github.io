        }

        &.#{$prefix}right {

                transform: rotate(90deg);
            }
        }

    display: flex;
    flex-direction: column;

    .#{$prefix}safari &,
    .#{$prefix}ie11 & {
        // Allows align: stretch to align items to the height of the tallest item
        // in an auto-heighted hbox layout.
